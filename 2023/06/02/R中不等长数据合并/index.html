R中不等长数据合并

# 创建 df1、df2 和 df3 数据框

df1 <- data.frame(value = 1:4)

df2 <- data.frame(value = 6:10)

df3 <- data.frame(value = 11:18)

df1$a=1

df2$a=2

df3$a=3

合并数据框

df4 <- rbind(df1,df2,df3)

library(reshape2)

# 删除共同变量

df5 <- dcast(df4,value~a)

df6 <- apply(df5, 2, function(x) { x[order(is.na(x))] })

df6[is.na(df6)] <- ""

df6 <- as.data.frame(df6)
